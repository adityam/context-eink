\setupmodule
  [
    \c!alternative=kindle,

  [
    \c!height=\einkdeviceparameter\c!height,
    \c!width=\einkdeviceparameter\c!width,
  ]

\setuppapersize[eink]

  [
    \c!location=\v!middle,
    \c!topspace=5mm,

    \c!height=\v!middle,

    \c!bottomdistance=\zeropoint,
    \c!bottom=\zeropoint,

    \c!leftmargin=1mm,
    \c!rightmargin=1mm,
    \c!header=\zeropoint,
    \c!footer=\zeropoint,
    \c!headerdistance=\zeropoint,
    \c!footerdistance=\zeropoint,
 ]

\setupwhitespace[\v!medium]
\setupindenting[\v!medium,\v!yes]

\setupsymbolset[text]

          ]

  \let\\=\crlf
\stopsetups

  \vfill\vfill\vfill
\stopsetups

\defineannotation
  [interlude]
  [
    \c!before=\setups{interlude::before},
    \c!after=\setups{interlude::after},
    \c!alternative=\v!paragraph,
    \c!color=,
    \c!align=\v!middle,
    \c!style={\interludefont\setupinterlinespace[\v!big]},
 ]
