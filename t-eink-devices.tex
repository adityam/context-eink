\startmodule eink-devices

\unprotect

\definenamespace
  [einkdevice]
  [   \c!type=module,
      \c!name=einkdevice,
   \c!command=\v!yes,
        setup=\v!list,
    \s!parent=einkdevice,
  ]

% Default size
\setupeinkdevice
  [
    \c!height=4.8in,
    \c!width=3.6in,
    \c!size=11pt,
  ]

% iRex devices: DR1000, DR800, iLiad
\defineeinkdevice
  [DR1000]
  [
    \c!height=8in,
    \c!width=6.25in,
    \c!size=12pt,
  ]

% Amazon Kindle devices: kindle, kindleDX, kindle-fire
% We scale the size to 95%, because that is how much of the screen kindle shows.
\defineeinkdevice
  [kindle]
  [
    \c!height=4.56in, %4.8in,
    \c!width=3.42in, %3.6in,
    \c!size=11pt,
  ]
